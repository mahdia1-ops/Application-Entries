import java.util.Scanner;

public class carClass {

    public static String carModel;
    public static String carBrand;
    public static double carMileage = 0.0;
    Scanner scnr = new Scanner(System.in);
    public void getCarInfo(){

        System.out.println("What is the car model?");
        String model = scnr.nextLine();
        this.carModel = model;

        System.out.println("What is the car brand");
        String brand = scnr.nextLine();
        this.carBrand = brand;
    }
    public void drive(){
        System.out.println("How much more have your driven since last time (miles please)");
        double mileage = scnr.nextDouble();
        this.carMileage =  this.carMileage + mileage;

    }
    public void printCarInfo(){
        System.out.println("Car Model: " + this.carModel + ". Car Brand: " + this.carBrand + ". Total Mileage: " + this.carMileage);
    }
}
